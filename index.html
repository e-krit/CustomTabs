Общий блок для табов - class="exampleTab js-tabsblock"

Блок для мобильного разрешения - class="exampleTab__mobileSelect js-mobile-tab-link"

Ссылка для смены таба - class="js-tab-link exampleTab__tab -active", так же у ссылки есть data-href с классом привязанным к открывающиму блоку, class="-active" показывает что данный блок открыт

Таб айтем - class="exampleTab__item js-tab__elem1 js-tab", class="js-tab__elem1" именной индификатор для связывания с ссылкой.

Далее необходимо написать js код для иницилизации табов.

/*TABS*/
$(document).on('click','.js-mobile-tab-link', function(e) {
	e.preventDefault();
	var $this = $(this),
		$block = $this.closest('.js-tabsblock');
		$tabLinkBlock = $block.find('.js-mobiletabs');

	$tabLinkBlock.toggleClass('visible');
	$this.toggleClass('active');
});

$(document).on('click','.js-tab-link', function(e) {
	e.preventDefault();
	var $this		= $(this),
		thisText	= $this.text(), 
		classBlock	= $this.data('href'),
		$block		= $this.closest('.js-tabsblock'),
		$mobileBlock = $block.find('.js-mobile-tab-link'),
		$tabLinkBlock = $block.find('.js-mobiletabs'),
		tabBlock	= $block.find('.js-tab');

	$('.js-tab-link').each(function(){
		$(this).removeClass('-active');
	});

	tabBlock.each(function() {
		$(this).hide();
	});

	$tabLinkBlock.removeClass('visible');
	$this.addClass('-active');
	$('.js-tab__' + classBlock).show();
	$mobileBlock.text(thisText).removeClass('active');
});
